    int fibonacci_recursive(int n) {
    if (n <= 1) {
        return n;
    } else {
        return fibonacci_recursive(n - 1) + fibonacci_recursive(n - 2);
    }
    }

    int main() {
    int n, i;
    printf("输入要计算的斐波那契数列的项数: ");

    printf("斐波那契数列: ");
